<template>
  <div id="container" class="min-h-screen bg-gradient-to-br from-gray-900
   via-gray-800 to-gray-900 py-12 px-4">
    <div class="max-w-4xl mx-auto">
      <div class="text-center mb-8">
        <h1 class="text-3xl font-bold text-white mb-2">API Tester</h1>
        <p class="text-gray-400">Enter your API endpoint and check the response</p>
      </div>
      <div class="bg-gray-800 rounded-2xl shadow-2xl overflow-hidden border border-gray-700">
        <div class="p-6">
          <div class="mb-6">
            <label for="input" class="block text-gray-300 font-semibold mb-2 text-sm">
              <textarea
                id="input"
                class="w-full px-4 py-3 bg-gray-900 border border-gray-700 rounded-lg
                 text-gray-200 focus:outline-none focus:border-green-500 focus:ring-2
                  focus:ring-green-500 transition-all duration-200 font-mono text-sm"
                placeholder="https://jsonplaceholder.typicode.com/posts/1"
              ></textarea>
            </label>
            <p class="text-gray-500 text-xs mt-2">Enter a valid API URL that returns JSON</p>
          </div>
          <div class="flex gap-4 mb-6">
            <button
              id="Check-btn"
              class="flex-1 px-6 py-3 bg-gradient-to-r from-green-500
               to-green-600 hover:from-green-600
               hover:to-green-700 text-white font-semibold rounded-lg
                transition-all duration-200 transform hover:scale-105 shadow-lg">
              🚀 Check API
            </button>
            <button
              id="Clear-btn"
              class="flex-1 px-6 py-3 bg-gradient-to-r from-red-500 to-red-600 hover:from-red-600
               hover:to-red-700 text-white font-semibold
                rounded-lg transition-all duration-200 transform hover:scale-105 shadow-lg">
              🗑️ Clear All
            </button>
          </div>
          <div class="border-t border-gray-700 pt-6">
            <div class="flex items-center gap-2 mb-4">
              <div class="w-2 h-2 bg-green-500 rounded-full animate-pulse"></div>
              <h3 class="text-gray-300 font-semibold text-sm">Response Preview</h3>
            </div>
            <div
              id="description"
              class="bg-gray-900 text-white rounded-lg border
               border-gray-700 p-4 min-h-[300px] max-h-[500px] overflow-y-auto">
            </div>
          </div>
        </div>
      </div>
    </div>
  </div>
</template>
<script>
import './assets/tailwind.css';

export default {
  name: 'App',
  mounted() {
    const descriptionEL = document.getElementById('description');
    const CheckEL = document.getElementById('Check-btn');
    const clearEL = document.getElementById('Clear-btn');
    const checkBtn = async () => {
      const inputEL = document.getElementById('input').value;
      try {
        const response = await fetch(inputEL);
        if (!response.ok) {
          throw new Error('we have Problem!(wrong Api)');
        }
        const data = await response.json();
        console.log(data);
        descriptionEL.insertAdjacentHTML('beforeend', `
          <p>${JSON.stringify(data, null, 2)}</p>
        `);
      } catch (error) {
        alert(error.message);
      }
    };
    CheckEL.addEventListener('click', checkBtn);
    clearEL.addEventListener('click', () => {
      descriptionEL.innerHTML = '';
    });
  },
};
</script>
